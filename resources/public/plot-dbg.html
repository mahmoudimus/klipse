{
  target: '#built-in-eval-function',
  data: [{
    // force the use of builtIn math
    graphType: 'polyline',
    fn: function (scope) {
      // scope.x = Number
      var x = scope.x
      return x * 2 + 4
    }
  }, {
    fnType: 'polar',
    graphType: 'polyline',
    r: function (scope) {
      // scope.theta = number
      var r0 = 0
      var a = 1
      var gamma = 0
      return r0 * Math.cos(scope.theta - gamma) +
        Math.sqrt(a * a - r0 * r0 * Math.pow(Math.sin(scope.theta - gamma), 2))
    }
  }]
}
